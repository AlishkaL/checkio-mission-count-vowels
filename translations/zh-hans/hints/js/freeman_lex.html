您只需将之前所有的提示连接起来，然后将代码封装成函数表达式即可。强烈建议对输入和输出数据使用类型提示。

以下是整个解决方案：

function countVowels(text: string): number {

    let count = 0;
    for(let char of text.toLowerCase()) {
        count += Number("aeiou".includes(char));
    }
    return count;
}

function countVowels(text: string): number {
    
    return text.replace(/[^aeiou]/gi, '').length;
